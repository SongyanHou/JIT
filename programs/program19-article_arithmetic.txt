x = myNode1 ++ myNode2 //UNION of two nodes

x = myNode1 ^^ myNode2 //INTERSECTION of two nodes

x = myNode1 ++ myNode2 over keywords //a UNION of keywords only

x = myNode1 ++ myNode2 over keywords, title //a UNION of keywords and title only

x = myNode1 ^^ myNode2 over keywords //an INTERSECTION of keywords only

x = myNode1 ^^ myNode2 over date, body, publisher, author //an INTERSECTION of date, body, publisher and author only
